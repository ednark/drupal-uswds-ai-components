USWDS Drupal Setup

Full module stack installation for a Drupal + USWDS government site.

# 1. Install base theme and paragraph components
composer require drupal/uswds
composer require drupal/uswds_paragraph_components

# 2. Install dependencies
composer require drupal/paragraphs drupal/entity_reference_revisions \
  drupal/viewsreference drupal/field_group drupal/twig_tweak

# 3. Enable modules
drush theme:enable uswds
drush pm:enable paragraphs entity_reference_revisions viewsreference field_group twig_tweak
drush pm:enable uswds_paragraph_components

# 4. Enable desired submodules (pick what you need)
drush pm:enable uswds_paragraph_components_accordion \
  uswds_paragraph_components_alert \
  uswds_paragraph_components_cards \
  uswds_paragraph_components_modal \
  uswds_paragraph_components_process_list \
  uswds_paragraph_components_step_indicator \
  uswds_paragraph_components_summary_box

# 5. Clear cache
drush cache:rebuild

# 6. Create a subtheme (recommended)
cp web/themes/contrib/uswds/examples/uswds_gulp_subtheme web/themes/custom/mytheme
drush config-set system.theme default mytheme
